select last_name||' '||first_name nume, email, salary,
       (select job_title from jobs where job_id = employees.job_id) post
        from employees
        where DEPARTMENT_ID = (select DEPARTMENT_ID
                                        from departments
                                        where department_name = 'Purchasing');

       (select job_title from jobs where job_id = employees.job_id) post
        from employees

                                        from departments
